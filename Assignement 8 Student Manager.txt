Assignment Name: Student Manager

JavaScript Code (Arrays & Objects + Average Calculation)

👉 Run in browser console (F12 → Console)

1. Storing Student Data
// Array of student objects
let students = [
    { name: "Preethi", marks: [85, 90, 88] },
    { name: "Shraddha", marks: [78, 82, 80] },
    { name: "Rahul", marks: [92, 87, 91] }
];

2. Function to Calculate Average
function calculateAverage(marks) {
    let sum = 0;

    for (let i = 0; i < marks.length; i++) {
        sum += marks[i];
    }

    return sum / marks.length;
}

3. Display Student Averages
students.forEach(function(student) {
    let avg = calculateAverage(student.marks);
    console.log(student.name + " Average Marks: " + avg);
});

4. Find Topper
let topper = "";
let highest = 0;

students.forEach(function(student) {
    let avg = calculateAverage(student.marks);

    if (avg > highest) {
        highest = avg;
        topper = student.name;
    }
});

console.log("Topper:", topper, "with average:", highest);
